//2.根据三个点求出三角形的面积
function AreaOfrange (x, y, z) {

    //声明两个空的向量
    var p1 = new THREE.Vector3();

    //根据两个顶点坐标坐标算出两个向量
    var v1 = x.clone().sub(y);
    var v2 = x.clone().sub(z);

    p1.crossVectors(v1, v2)
    return p1.length() / 2
}

    area += AreaOfrange(p1, p2, p3)
}
